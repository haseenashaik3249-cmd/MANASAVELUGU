function analyzeCode() {
    const code = document.getElementById("codeInput").value;
    const loading = document.getElementById("loading");
    const resultDiv = document.getElementById("result");

    if (code.trim() === "") {
        alert("Please paste some code first!");
        return;
    }

    loading.innerText = "Analyzing code using AI...";
    resultDiv.style.display = "none";

    setTimeout(() => {
        document.getElementById("bugs").innerHTML =
            "<li>Unused variable detected</li><li>Possible null reference error</li>";

        document.getElementById("performance").innerHTML =
            "<li>Use caching or memoization</li><li>Avoid nested loops if possible</li>";

        document.getElementById("security").innerHTML =
            "<li>Avoid hardcoded credentials</li><li>Sanitize user inputs</li>";

        document.getElementById("optimizedCode").innerText =
`// Optimized Example
function sum(a, b) {
    if (a == null || b == null) return 0;
    return a + b;
}`;

        loading.innerText = "";
        resultDiv.style.display = "block";

    }, 2000);
}
